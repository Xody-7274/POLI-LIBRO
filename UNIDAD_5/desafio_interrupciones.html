// Llamada al sistema en Linux x86
void syscall() {
    asm("mov $1, %eax\n"
        "mov $0, %ebx\n"
        "");
}

// ISR para interrupción de timer en AVR
#include <avr/interrupt.h>

ISR() {
    PORTB ^= (1 << PB0);
}
